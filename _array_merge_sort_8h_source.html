45 void ArrayMergeSort<T>::MergeCore(IArrayCollection<T>& list, int low, int mid, int high)
46 {
47  int i, j;
48  auto n1 = mid - low + 1;
49  auto n2 = high - mid;
50 
51  /* create temp arrays */
52  T* l = new T[n1];
53  T* r = new T[n2];
54 
55  /* Copy data to temp arrays L[] and R[] */
56  for (i = 0; i < n1; i++)
57  l[i] = list[low + i];
58  for (j = 0; j < n2; j++)
59  r[j] = list[mid + 1 + j];
60 
61  /* Merge the temp arrays back into arr[l..r]*/
62  i = 0; // Initial index of first subarray
63  j = 0; // Initial index of second subarray
64  auto k = low; // Initial index of merged subarray
65  while (i < n1 && j < n2)
66  {
67  if (_comparePredicate(r[j], l[i]))
68  {
69  list[k] = l[i];
70  i++;
71  }
72  else
73  {
74  list[k] = r[j];
75  j++;
76  }
77  k++;
78  }
79 
80  /* Copy the remaining elements of L[], if there
81  are any */
82  while (i < n1)
83  {
84  list[k] = l[i];
85  i++;
86  k++;
87  }
88 
89  /* Copy the remaining elements of R[], if there
90  are any */
91  while (j < n2)
92  {
93  list[k] = r[j];
94  j++;
95  k++;
96  }
97 
98  delete[] l;
